# Astro Framework Features & Theme Capabilities

## üöÄ Introduction to Astro

Astro is a modern web framework designed for building fast, content-focused websites. It combines the best of static site generation with modern JavaScript frameworks, allowing developers to create high-performance websites with minimal JavaScript.

### Why Choose Astro?

- **Zero JavaScript by Default**: Ships no client-side JavaScript unless you need it
- **Islands Architecture**: Interactive components are opt-in and isolated
- **Static Site Generation**: Lightning-fast loading times
- **Multi-Framework Support**: Use React, Vue, Svelte, or vanilla JavaScript
- **Content Collections**: Type-safe content management
- **Edge Deployment**: Deploy to any edge network worldwide

### 3. Search Functionality

```typescript
// src/scripts/search.js
import Fuse from 'fuse.js';

class PixelSearch {
  constructor() {
    this.searchIndex = [];
    this.fuse = null;
    this.init();
  }

  async init() {
    // Load search index
    const response = await fetch('/search-index.json');
    const data = await response.json();

    this.searchIndex = data;
    this.fuse = new Fuse(this.searchIndex, {
      keys: ['title', 'description', 'content', 'tags'],
      threshold: 0.3,
      includeScore: true,
      includeMatches: true,
    });
  }

  search(query) {
    if (!this.fuse || !query) return [];

    const results = this.fuse.search(query);
    return results.map(result => ({
      ...result.item,
      score: result.score,
      matches: result.matches,
    }));
  }
}
```

### 1. Dynamic Color Themes

```typescript
// src/utils/theme.js
export class ThemeManager {
  constructor() {
    this.currentTheme = localStorage.getItem('pixel-theme') || 'dark';
    this.applyTheme();
  }

  applyTheme() {
    const root = document.documentElement;
    const themes = {
      dark: {
        '--bg-primary': '#0a0a0a',
        '--bg-secondary': '#1a1a1a',
        '--text-primary': '#ffffff',
        '--accent-color': '#00ffff',
      },
      light: {
        '--bg-primary': '#ffffff',
        '--bg-secondary': '#f0f0f0',
        '--text-primary': '#000000',
        '--accent-color': '#00ff00',
      },
      retro: {
        '--bg-primary': '#1a1a00',
        '--bg-secondary': '#2a2a00',
        '--text-primary': '#ffff00',
        '--accent-color': '#ff00ff',
      },
    };

    const theme = themes[this.currentTheme];
    Object.entries(theme).forEach(([key, value]) => {
      root.style.setProperty(key, value);
    });

    localStorage.setItem('pixel-theme', this.currentTheme);
  }

  setTheme(themeName) {
    this.currentTheme = themeName;
    this.applyTheme();
  }
}
```

<script>
  // Progressive enhancement - adds modal functionality
  class GalleryModal {
    constructor(gallery) {
      this.gallery = gallery;
      this.items = gallery.querySelectorAll('.pixel-gallery-item');
      this.setupModal();
      this.bindEvents();
    }

    setupModal() {
      this.modal = document.createElement('div');
      this.modal.className = 'pixel-modal';
      this.modal.innerHTML = `
        <div class="pixel-modal-content">
          <img src="" alt="" class="pixel-modal-image">
          <div class="pixel-modal-caption"></div>
          <button class="pixel-modal-close">&times;</button>
          <button class="pixel-modal-prev">‚Äπ</button>
          <button class="pixel-modal-next">‚Ä∫</button>
        </div>
      `;
      document.body.appendChild(this.modal);
    }

    bindEvents() {
      this.items.forEach((item, index) => {
        item.addEventListener('click', () => this.openModal(index));
      });

      this.modal.querySelector('.pixel-modal-close').addEventListener('click', () => this.closeModal());
      this.modal.querySelector('.pixel-modal-prev').addEventListener('click', () => this.navigate(-1));
      this.modal.querySelector('.pixel-modal-next').addEventListener('click', () => this.navigate(1));

      document.addEventListener('keydown', (e) => {
        if (e.key === 'Escape') this.closeModal();
        if (e.key === 'ArrowLeft') this.navigate(-1);
        if (e.key === 'ArrowRight') this.navigate(1);
      });
    }

    openModal(index) {
      this.currentIndex = index;
      this.updateModal();
      this.modal.classList.add('pixel-modal-active');
      document.body.style.overflow = 'hidden';
    }

    closeModal() {
      this.modal.classList.remove('pixel-modal-active');
      document.body.style.overflow = '';
    }

    navigate(direction) {
      this.currentIndex = (this.currentIndex + direction + this.items.length) % this.items.length;
      this.updateModal();
    }

    updateModal() {
      const item = this.items[this.currentIndex];
      const img = item.querySelector('img');
      const caption = item.querySelector('.pixel-gallery-caption');

      this.modal.querySelector('.pixel-modal-image').src = img.src;
      this.modal.querySelector('.pixel-modal-image').alt = img.alt;
      this.modal.querySelector('.pixel-modal-caption').textContent = caption?.textContent || '';
    }
  }

  // Initialize gallery when DOM is ready
  document.addEventListener('DOMContentLoaded', () => {
    const gallery = document.querySelector('.pixel-gallery');
    if (gallery) new GalleryModal(gallery);
  });
</script>
